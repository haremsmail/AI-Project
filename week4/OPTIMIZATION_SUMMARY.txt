        self.running = False
        self.particles, self.iteration, self.history = [], 0, []
        self.global_best = (0, 0, float('inf'))
        self.history_list.delete(0, tk.END)
        self.info.config(text="Iter: 0 | Fit: -")
        self.canvas.delete("all")
